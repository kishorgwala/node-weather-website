import { render, fireEvent } from '@testing-library/react';
import EmeaRecommendation from './EmeaRecommendation';

describe('EmeaRecommendation', () => {
  it('renders without crashing', () => {
    render(<EmeaRecommendation />);
  });

  it('renders the file type select menu', () => {
    const { getByLabelText } = render(<EmeaRecommendation />);
    const select = getByLabelText(/File Type/i);
    expect(select).toBeInTheDocument();
  });

  it('changes value when option is selected', () => {
    const { getByLabelText } = render(<EmeaRecommendation />);
    const select = getByLabelText(/File Type/i);
    fireEvent.change(select, { target: { value: 'SIX_Tks_Equity' } });
    expect(select.value).toBe('SIX_Tks_Equity');
  });

  // Add more tests for each file type and the expected component to be rendered
  it('renders DisplaySixTksEq when SIX_Tks_Equity is selected', () => {
    const { getByLabelText, getByTestId } = render(<EmeaRecommendation />);
    const select = getByLabelText(/File Type/i);
    fireEvent.change(select, { target: { value: 'SIX_Tks_Equity' } });
    const displayComponent = getByTestId('display-six-tks-eq');
    expect(displayComponent).toBeInTheDocument();
  });
});
